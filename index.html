import { getFirestore, collection, addDoc, getDocs, deleteDoc, doc } from "https://www.gstatic.com/firebasejs/12.0.0/firebase-firestore.js"; const db = getFirestore(app); // إضافة أصل window.addAsset = async function () { const name = document.getElementById("assetName").value; const code = document.getElementById("assetCode").value; await addDoc(collection(db, "assets"), { name, code, createdAt: Date.now() }); loadAssets(); }; // تحميل الأصول async function loadAssets() { const querySnapshot = await getDocs(collection(db, "assets")); let rows = ""; querySnapshot.forEach((d) => { const data = d.data(); rows += ` ${data.name} ${data.code} `; }); document.getElementById("assetsTable").innerHTML = rows; } // حذف أصل window.deleteAsset = async function (id) { await deleteDoc(doc(db, "assets", id)); loadAssets(); };
